Since SINs may be used as input data for some applications, I have translated the code that generates a number that passes the Luhn Algorithm in Wikipedia to something that can be put into into Selenium’s user-extensions.js.

Selenium.prototype.getRandomSIN = function() {
    var length = 9;
    var digits = new Array();
    for(var i = 0; i &lt; length - 1; ++i) {         digits[i] = Math.floor(Math.random() * 9)     }     var sum = 0;     var alt = true;     for(var i = length - 2; i &gt;= 0; --i) {
        if (alt) {
            var temp = digits[i];
            temp *= 2;
            if (temp &gt; 9) {
                temp -= 9;
            }
            sum += temp;
        } else {
            sum += digits[i];
        }
        alt = !alt;
    }

    var modulo = sum % 10;
    if(modulo &gt; 0) {
        digits[length-1] = 10 - modulo;
    }
    return digits.join("");
};
